DROP DATABASE IF EXISTS todo;

CREATE DATABASE todo DEFAULT CHARACTER SET utf8;

GRANT ALL ON todo.* TO 'todo' @'localhost' IDENTIFIED BY 'todo';
GRANT ALL ON todo.* TO 'todo' @'127.0.1' IDENTIFIED BY 'todo';

USE todo;

DROP TABLE IF EXISTS new;

CREATE TABLE new(
	new_id INTEGER NOT NULL AUTO_INCREMENT,
	task VARCHAR(128),
	completed INTEGER,
	new INTEGER,
	PRIMARY KEY(new_id),
	INDEX (task)
	)ENGINE=INNODB DEFAULT CHARACTER SET=utf8;
